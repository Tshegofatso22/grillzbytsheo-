const express = require('express'); const router = express.Router(); const Product = require('../models/Product'); // Get all products router.get('/', async (req, res) => { try { const products = await Product.find(); res.json(products); } catch (err) { res.status(500).send(err); } }); // Get product by ID router.get('/:id', async (req, res) => { try { const product = await Product.findById(req.params.id); res.json(product); } catch (err) { res.status(500).send(err); } }); module.exports = router;
